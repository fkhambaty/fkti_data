📊 Monitoring

-- System tables for monitoring
SELECT * FROM system.metrics;        -- Current metrics
SELECT * FROM system.events;         -- Cumulative counters
SELECT * FROM system.query_log;      -- Query history
SELECT * FROM system.parts;          -- Data parts info
SELECT * FROM system.replicas;       -- Replication status

-- Useful monitoring queries
SELECT
    database,
    table,
    formatReadableSize(sum(bytes_on_disk)) AS size,
    sum(rows) AS total_rows,
    count() AS parts
FROM system.parts
WHERE active
GROUP BY database, table
ORDER BY sum(bytes_on_disk) DESC;

The most popular monitoring setup: export ClickHouse metrics to Prometheus and visualize in Grafana. ClickHouse has a built-in Prometheus endpoint at /metrics.

💾 Backups

-- Built-in backup (ClickHouse 22.8+)
BACKUP TABLE events TO Disk('backups', 'events_backup.zip');

-- Restore
RESTORE TABLE events FROM Disk('backups', 'events_backup.zip');

-- Backup entire database
BACKUP DATABASE default TO S3('https://bucket.s3.amazonaws.com/backups/');

⚠️ Backup Strategy

  • Daily incremental backups to S3/GCS
  • Weekly full backups
  • Test restores monthly
  • Replication is NOT a backup!

🔐 Security

-- Create users with specific permissions
CREATE USER analyst IDENTIFIED BY 'strong_password';
GRANT SELECT ON analytics.* TO analyst;

CREATE USER etl_service IDENTIFIED BY 'service_password';
GRANT INSERT ON raw_data.* TO etl_service;

-- Row-level security
CREATE ROW POLICY us_only ON events
FOR SELECT
USING country = 'US'
TO us_analyst;

-- Load from S3
INSERT INTO events
SELECT * FROM s3(
    'https://bucket.s3.amazonaws.com/data/*.parquet',
    'AWS_KEY', 'AWS_SECRET',
    'Parquet'
);

-- Stream from Kafka
CREATE TABLE events_kafka (
    event_time DateTime,
    user_id UInt64,
    event_type String
) ENGINE = Kafka
SETTINGS
    kafka_broker_list = 'kafka:9092',
    kafka_topic_list = 'events',
    kafka_group_name = 'clickhouse_consumer',
    kafka_format = 'JSONEachRow';
