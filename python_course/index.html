Example 2: Sales Data Analysis

import pandas as pd
import matplotlib.pyplot as plt

# Monthly sales data
sales_data = {
    'Month': ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'],
    'Sales': [15000, 18000, 22000, 19000, 21000, 24000],
    'Expenses': [12000, 13000, 15000, 14000, 15000, 16000]
}

df = pd.DataFrame(sales_data)

# Calculate profit
df['Profit'] = df['Sales'] - df['Expenses']

# Key insights
print("Total Sales:", df['Sales'].sum())
print("Total Profit:", df['Profit'].sum())
print("Average Monthly Profit:", df['Profit'].mean())
print("Best Month:", df.loc[df['Profit'].idxmax(), 'Month'])

# Growth rate
df['Growth'] = df['Sales'].pct_change() * 100
print("\nMonth-over-Month Growth:")
print(df[['Month', 'Growth']].dropna())

# Create visualization
plt.figure(figsize=(10, 6))
plt.plot(df['Month'], df['Sales'], marker='o', label='Sales')
plt.plot(df['Month'], df['Expenses'], marker='s', label='Expenses')
plt.xlabel('Month')
plt.ylabel('Amount ($)')
plt.title('Sales vs Expenses')
plt.legend()
plt.grid(True)
plt.show()
ğŸ” Business Insights:

This analysis helps identify profitable months, track growth trends, and make data-driven decisions!

Example 3: Simple Machine Learning

Predict house prices based on size!

from sklearn.linear_model import LinearRegression
import numpy as np
import matplotlib.pyplot as plt

# House data: size and price
sizes = np.array([800, 1000, 1200, 1400, 1600, 1800, 2000]).reshape(-1, 1)
prices = np.array([150000, 180000, 210000, 240000, 270000, 300000, 330000])

# Create and train model
model = LinearRegression()
model.fit(sizes, prices)

# Predict price for 1500 sq ft house
predicted_price = model.predict([[1500]])
print(f"Predicted price for 1500 sq ft: ${predicted_price[0]:,.0f}")

# Accuracy
score = model.score(sizes, prices)
print(f"Model accuracy: {score * 100:.2f}%")

# Visualize
plt.scatter(sizes, prices, color='blue', label='Actual Data')
plt.plot(sizes, model.predict(sizes), color='red', label='Prediction')
plt.xlabel('House Size (sq ft)')
plt.ylabel('Price ($)')
plt.title('House Price Prediction')
plt.legend()
plt.show()
ğŸ¤– This is Machine Learning!

The computer learned the pattern from existing data and can now predict prices for new houses!
